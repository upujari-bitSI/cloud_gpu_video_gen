# =====================================================================

# (Linux + CUDA 13, Python 3.11, April 2026).
#
# IMPORTANT install order: see setup.sh in the repo root. TTS's deps will
# pull in newer transformers/huggingface_hub if installed unconstrained,
# which then breaks the pinned diffusers below. The setup.sh script
# installs TTS first then pins the rest, OR you can run:

# ---- Core LLM -------------------------------------------------------
anthropic>=0.40.0
openai>=1.50.0
python-dotenv>=1.0.0

# ---- Image generation (Flux primary, SDXL fallback) -----------------
# Flux.1 [schnell] requires diffusers>=0.30. The known-good Flux stack:
#   diffusers 0.31.0 + transformers 4.45.2 + huggingface_hub 0.25.2.
# SDXL still works under these versions for the fallback path.
# Compatibility constraints learned the hard way:
#   * diffusers <0.28 imports cached_download (removed in huggingface_hub>=0.26)
#   * diffusers >=0.32 has new attention_dispatch breaking on older torch
#   * transformers 5.x requires huggingface_hub>=1.5 (TTS pulls this in if
#     not pinned), so install order in setup.sh matters
#   * torchvision must match torch ABI (upgrade together)
torch==2.11.0
torchvision==0.26.0
diffusers==0.31.0
huggingface_hub==0.25.2
transformers==4.45.2
accelerate==0.34.2
safetensors>=0.4.0
sentencepiece>=0.2.0
protobuf>=4.25.0
xformers; platform_system != "Darwin"

# ---- TTS ------------------------------------------------------------

# much more natural than the original Tacotron2-DDC). Coqui kept as
# fallback. ElevenLabs is optional (paid API).

soundfile>=0.12.0

elevenlabs>=1.5.0

# ---- Video processing ----------------------------------------------
